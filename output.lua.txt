local function toUnicode(string)
    local codepoints = "utf8.char("
    for _i, v in utf8.codes(string) do
        codepoints = codepoints .. v .. ', '
    end
    return codepoints:sub(1, -3) .. ')'
end

local function GetFullPathOfAnInstance(instance)
    if not instance then return "nil" end
    local name = instance.Name
    local head = (#name > 0 and '.' .. name) or "['']"

    if not instance.Parent and instance ~= game then
        return head .. " --[[ PARENTED TO NIL OR DESTROYED ]]"
    end

    if instance == game then
        return "game"
    elseif instance == workspace then
        return "workspace"
    else
        local success, result = pcall(game.GetService, game, instance.ClassName)
        if result then
            head = ':GetService("' .. instance.ClassName .. '")'
        elseif instance == client then
            head = '.LocalPlayer' 
        else
            local nonAlphaNum = name:gsub('[%w_]', '')
            local noPunct = nonAlphaNum:gsub('[%s%p]', '')

            if tonumber(name:sub(1, 1)) or (#nonAlphaNum ~= 0 and #noPunct == 0) then
                head = '["' .. name:gsub('"', '\\"'):gsub('\\', '\\\\') .. '"]'
            elseif #nonAlphaNum ~= 0 and #noPunct > 0 then
                head = '[' .. toUnicode(name) .. ']'
            end
        end
    end
    return GetFullPathOfAnInstance(instance.Parent) .. head
end

-- Functionality
local function FindRemote(remote, args)
    for i, v in ipairs(remotes) do
        if v == remote then
            if not table.find(unstacked, remote) then
                return i
            else
                -- Check args for unstacked
                local match = true
                if #remoteArgs[i] == #args then
                    for k, arg in pairs(args) do
                        if remoteArgs[i][k] ~= arg then match = false break end
                    end
                else match = false end
                if match then return i end
            end
        end
    end
    return nil
end

local function convertTableToString(args)
    local result = ""
    for i, v in pairs(args) do
        local val = tostring(v)
        if type(v) == "string" then val = '"' .. v .. '"'
        elseif typeof(v) == "Instance" then val = GetFullPathOfAnInstance(v)
        elseif type(v) == "table" then val = "{" .. convertTableToString(v) .. "}"
        end
        result = result .. (type(i) == "number" and "" or '["'..tostring(i)..'"] = ') .. val .. ", "
    end
    return result:sub(1, -3)
end

local function addToList(event, remote, ...)
    if not remote then return end
    local args = {...}
    local i = FindRemote(remote, args)

    if not i then
        table.insert(remotes, remote)
        table.insert(remoteArgs, args)
        
        local rButton = clone(RemoteButton)
        rButton.Parent = RemoteScrollFrame
        rButton.Visible = true
        rButton.RemoteName.Text = remote.Name
        rButton.RemoteIcon.Image = event and eventImage or functionImage
        
        buttonOffset = buttonOffset + 35
        rButton.Position = UDim2.new(0, 10, 0, buttonOffset)
        RemoteScrollFrame.CanvasSize = UDim2.new(0, 0, 0, buttonOffset + 40)
        
        local index = #remotes
        remoteButtons[index] = rButton.Number
        
        rButton.MouseButton1Click:Connect(function()
            InfoFrame.Visible = true
            Code.Text = GetFullPathOfAnInstance(remote) .. (event and ":FireServer(" or ":InvokeServer(") .. convertTableToString(args) .. ")"
            lookingAt = remote
            lookingAtArgs = args
        end)
    else
        remoteButtons[i].Text = tostring(tonumber(remoteButtons[i].Text) + 1)
        remoteArgs[i] = args
    end
end

-- Hooking (Improved for Mobile Executors)
local function safe_hook(obj, func_name, callback)
    local success, old = pcall(function()
        return hookfunction(obj[func_name], function(self, ...)
            local result = {callback(self, ...)}
            if #result > 0 then return unpack(result) end
            return old(self, ...)
        end)
    end)
    return success
end

-- Hook RemoteEvent
local OldFire
OldFire = hookfunction(Instance.new("RemoteEvent").FireServer, function(self, ...)
    if not checkcaller() then
        pcall(addToList, true, self, ...)
    end
    return OldFire(self, ...)
end)

-- Hook RemoteFunction
local OldInvoke
OldInvoke = hookfunction(Instance.new("RemoteFunction").InvokeServer, function(self, ...)
    if not checkcaller() then
        pcall(addToList, false, self, ...)
    end
    return OldInvoke(self, ...)
end)

-- Metatable Hook (__namecall)
local mt = getrawmetatable(game)
local oldNamecall = mt.__namecall
setreadonly(mt, false)

mt.__namecall = newcclosure(function(self, ...)
    local method = getnamecallmethod()
    if not checkcaller() then
        if method == "FireServer" then
            pcall(addToList, true, self, ...)
        elseif method == "InvokeServer" then
            pcall(addToList, false, self, ...)
        end
    end
    return oldNamecall(self, ...)
end)

setreadonly(mt, true)
